from scipy.stats import chi2_contingency
import pandas as pd

# Filter to top 10 complaint types and top 5 boroughs for simplicity
top_complaints = df['Complaint Type'].value_counts().head(10).index
top_boroughs = df['Borough'].value_counts().head(5).index
filtered_df = df[df['Complaint Type'].isin(top_complaints) & df['Borough'].isin(top_boroughs)]

# Create the contingency table
contingency_table = pd.crosstab(filtered_df['Complaint Type'], filtered_df['Borough'])

# Apply chi-square test
chi2, p, dof, expected = chi2_contingency(contingency_table)

print("Contingency Table:\n", contingency_table)

print("Chi-Square Statistic:", chi2)
print("P-value:", p)

# Interpretation
if p < 0.05:
    print("There is a statistically significant relationship between complaint type and borough.")
else:
    print("There is no statistically significant relationship between complaint type and borough.")
